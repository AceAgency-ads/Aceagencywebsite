    2. **Update `src/app/layout.tsx`** to load all fonts and apply CSS variables to the html element:

    ```tsx
    import type { Metadata } from 'next'
    import localFont from 'next/font/local'
    import { Poppins, Red_Hat_Display, Anton } from 'next/font/google'
    import './globals.css'

    // Glacial Indifference - local font (or fallback)
    const glacialIndifference = localFont({
      src: [
        { path: './fonts/GlacialIndifference-Regular.woff2', weight: '400', style: 'normal' },
        { path: './fonts/GlacialIndifference-Bold.woff2', weight: '700', style: 'normal' },
      ],
      variable: '--font-glacial',
      display: 'swap',
    })

    // Poppins - Google Font
    const poppins = Poppins({
      subsets: ['latin', 'latin-ext'],
      weight: ['400', '500'],
      variable: '--font-poppins',
      display: 'swap',
    })

    // Red Hat Display - Google Font
    const redHatDisplay = Red_Hat_Display({
      subsets: ['latin', 'latin-ext'],
      weight: ['400'],
      variable: '--font-redhat',
      display: 'swap',
    })

    // Anton - Google Font
    const anton = Anton({
      subsets: ['latin', 'latin-ext'],
      weight: ['400'],
      variable: '--font-anton',
      display: 'swap',
    })

    export const metadata: Metadata = {
      title: 'AceAgency',
      description: 'Full-service digital marketing agency in Bucharest, Romania',
    }

    export default function RootLayout({
      children,
    }: {
      children: React.ReactNode
    }) {
      return (
        <html
          lang="ro"
          className={`${glacialIndifference.variable} ${poppins.variable} ${redHatDisplay.variable} ${anton.variable}`}
        >
          <body className="antialiased">{children}</body>
        </html>
      )
    }
    ```

    NOTE: The `lang="ro"` is temporary -- Plan 02 will move this to the locale-specific layout where the lang attribute is set dynamically based on the current locale.
